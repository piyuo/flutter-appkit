import 'package:flutter_test/flutter_test.dart';
import 'package:libcli/log.dart';
import 'package:libcli/src/log/global-exception-handler.txt';
import 'package:mockito/mockito.dart';

class MockBuildContext extends Mock implements BuildContext {}

void main() {
  group('[global-exception-handler]', () {
    test('should sendToGlobalExceptionHanlder', () {
      globalExceptionHandler = (BuildContext context, dynamic e, StackTrace s) {
        expect(e, 'mock');
        expect(s, isNotNull);
      };

      try {
        throw 'mock';
      } catch (e, s) {
        throwToGlobalExceptionHandler(MockBuildContext(), e, s);
      }
    });
  });
}
